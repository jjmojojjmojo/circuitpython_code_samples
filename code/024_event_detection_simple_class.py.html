from setup import check, led, rgb
import time

class AButtonEvents:
    _debounce = 0.2
    
    def __init__(self):
        self.state = False
        self.checkin = time.monotonic()

    def press(self):
        print("'A' button pressed")
        
    def release(self):
        print("'A' button released")
    
    def update(self):
        if time.monotonic() - self.checkin > self._debounce:
            if not self.state and check("A"):
                self.press()
                
            if self.state and not check("A"):
                self.release()
                
            self.state = check("A")
            
            self.checkin = time.monotonic()

    _debounce = 0.2

                self.press()

b = BButtonEvents()
a = AButtonEvents()

while True:
    a.update()
